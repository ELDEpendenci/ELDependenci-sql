001package chu77.eldependenci.plugin;
002
003import chu77.eldependenci.sql.config.Dbconfig;
004import chu77.eldependenci.sql.manager.SQLDataSourceManager;
005import chu77.eldependenci.sql.SQLService;
006import com.ericlam.mc.eld.ELDLifeCycle;
007import org.bukkit.plugin.java.JavaPlugin;
008
009import javax.inject.Inject;
010
011public class Lifecycle implements ELDLifeCycle {
012
013    @Inject
014    private Dbconfig dbconfig;
015
016    @Inject
017    private SQLDataSource sqlDataSource;
018
019    @Override
020    public void onEnable(JavaPlugin javaPlugin) {
021        if (dbconfig.enable){ // 如果 sql 已被啟用
022            javaPlugin.getLogger().info("正在初始化 mysql....");
023            ((SQLDataSourceManager)sqlDataSource).initialize(); // 初始化sql
024        }
025    }
026
027    @Override
028    public void onDisable(JavaPlugin javaPlugin) {
029    }
030}
