001package chu77.eldependenci.sql.sqlmanager;
002
003import chu77.eldependenci.sql.config.Dbconfig;
004import chu77.eldependenci.sql.SQLService;
005import com.zaxxer.hikari.HikariConfig;
006import com.zaxxer.hikari.HikariDataSource;
007import org.sql2o.Sql2o;
008
009import javax.inject.Inject;
010import javax.sql.DataSource;
011import java.sql.Connection;
012import java.sql.SQLException;
013
014
015public class SQLDataSourceManager implements SQLDataSource {
016
017    private static DataSource dataSource;
018
019    @Inject
020    private Dbconfig db;
021
022    private Sql2o sql2o;
023
024    public void initialize(){
025        if (!db.enable) throw new IllegalStateException("SQL 已被禁用，請在 db.yml 啟用它");
026        HikariConfig config = new HikariConfig();
027        String host = db.host;
028        String port = db.port + "";
029        String database = db.database;
030        String username = db.username;
031        String password = db.password;
032        String poolname = db.pool.name;
033        int minsize = db.pool.minSize;
034        int maxsize = db.pool.maxSize;
035        boolean SSL = db.useSSL;
036        long connectionTimeout = db.pool.connectionTimeout;
037        long idleTimeout = db.pool.idleTimeout;
038        long maxLifeTime = db.pool.maxLifeTime;
039        String jdbc = "jdbc:mysql://" + host + ":" + port + "/" + database + "?" + "useSSL=" + SSL;
040        config.setJdbcUrl(jdbc);
041        config.setPoolName(poolname);
042        config.setMaximumPoolSize(maxsize);
043        config.setMinimumIdle(minsize);
044        config.setUsername(username);
045        config.setPassword(password);
046        config.setConnectionTimeout(connectionTimeout);
047        config.setIdleTimeout(idleTimeout);
048        config.setMaxLifetime(maxLifeTime);
049        config.addDataSourceProperty("cachePrepStmts", true);
050        config.addDataSourceProperty("useServerPrepStmts", true);
051        config.addDataSourceProperty("prepStmtCacheSize", 250);
052        config.addDataSourceProperty("prepStmtCacheSqlLimit", 2048);
053        config.addDataSourceProperty("characterEncoding", "utf8");
054        dataSource = new HikariDataSource(config);
055        sql2o = new Sql2o(dataSource);
056    }
057
058    @Override
059    public Connection getConnection() throws SQLException {
060        if (!db.enable) throw new IllegalStateException("SQL 已被禁用，請在 db.yml 啟用它");
061        return dataSource.getConnection();
062    }
063
064    @Override
065    public DataSource getDataSource() {
066        if (!db.enable) throw new IllegalStateException("SQL 已被禁用，請在 db.yml 啟用它");
067        return dataSource;
068    }
069
070    @Override
071    public Sql2o getSql2o(){
072        if (!db.enable) throw new IllegalStateException("SQL 已被禁用，請在 db.yml 啟用它");
073        return sql2o;
074    }
075}
